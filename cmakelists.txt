cmake_minimum_required(VERSION 3.15)

project(CollectionsLibrary)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

enable_testing()

add_subdirectory(external/googletest)

include_directories(
    src/DynamicArray
    "src/Linkedlist"
    src/HashMap
    src/RedisLite
    src/Exceptions
)

# Dynamic Array
add_executable(
    DynamicArray
    src/DynamicArray/Dynamicmain.cpp
)

# Linked List
add_executable(
    LinkedList
    src/LinkedList/Linkedmain.cpp
)

# HashMap
add_executable(
    HashMap
    src/HashMap/hmap.cpp
)

# Redis Lite
add_executable(
    RedisLite
    src/RedisLite/Redismain.cpp
)

# GoogleTest
file(GLOB_RECURSE TEST_SOURCES
    tests/*.cpp
)

add_executable(
    CollectionTests
    ${TEST_SOURCES}
)

target_link_libraries(
    CollectionTests
    PRIVATE
    gtest
    gtest_main
)

include(GoogleTest)
gtest_discover_tests(CollectionTests)
