cmake_minimum_required(VERSION 3.15)

project(CollectionsLibrary)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

enable_testing()

add_subdirectory(external/googletest)

# ==================================
# Collections Library
# ==================================

add_library(Collections INTERFACE)

target_include_directories(Collections INTERFACE
    ${CMAKE_CURRENT_SOURCE_DIR}/include
    ${CMAKE_CURRENT_SOURCE_DIR}/include/DynamicArray
    ${CMAKE_CURRENT_SOURCE_DIR}/include/LinkedList
    ${CMAKE_CURRENT_SOURCE_DIR}/include/HashMap
    ${CMAKE_CURRENT_SOURCE_DIR}/include/RedisLite
    ${CMAKE_CURRENT_SOURCE_DIR}/include/Exceptions
)

# ==================================
# src
# ==================================

add_executable(DynamicArray
    src/DynamicArray/Dynamicmain.cpp
)

target_link_libraries(DynamicArray PRIVATE Collections)

add_executable(LinkedList
    src/LinkedList/Linkedmain.cpp
)

target_link_libraries(LinkedList PRIVATE Collections)

add_executable(HashMap
    src/HashMap/hmap.cpp
)

target_link_libraries(HashMap PRIVATE Collections)

add_executable(RedisLite
    src/RedisLite/Redismain.cpp
)

target_link_libraries(RedisLite PRIVATE Collections)

# ==================================
# Tests
# ==================================

file(GLOB_RECURSE TEST_SOURCES
    tests/*.cpp
)

add_executable(CollectionTests
    ${TEST_SOURCES}
)

target_link_libraries(CollectionTests
    PRIVATE
    Collections
    gtest
    gtest_main
)

include(GoogleTest)
gtest_discover_tests(CollectionTests)
